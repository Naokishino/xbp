ステッピングモーター

  1. const int DIR = 8;
  2. const int STEP = 9;
  4. void setup() {
  5.   pinMode(DIR, OUTPUT);
  6.   pinMode(STEP, OUTPUT);
  7.   digitalWrite(DIR, LOW);
  8.   digitalWrite(STEP, LOW);
  9.   Serial.begin(9600);
  10. }
  12. void loop() {
  13.   int val=0;
  14.   val=analogRead(1);
  15.   Serial.println(val);
  16.   delay(50);
  17.  if(val<200){
  18.       clockwise(5000);
  19.     }else{
  20.       
  21.        counter_clockwise(5000);
  22.     }
  23.   
  24.   }
  26. void clockwise(int delaytime){
  27.   digitalWrite(DIR, HIGH);//HIGHは時計回り
  28.   digitalWrite(STEP, HIGH);
  29.   delayMicroseconds(delaytime);
  30.   digitalWrite(STEP, LOW);
  31.   delayMicroseconds(delaytime);
  33. }
  35. void counter_clockwise(int delaytime){
  37.   digitalWrite(DIR, LOW);//はLOWは反時計回り
  38.   digitalWrite(STEP, HIGH);
  39.   delayMicroseconds(delaytime);
  40.   digitalWrite(STEP, LOW);
  41.   delayMicroseconds(delaytime);
  42. }
ifなどを使うことで明るさの数値からうごかしたり止めたりすることができる
